CREATE TABLE Venda (
    idVenda INT AUTO_INCREMENT PRIMARY KEY,
    valorTotal DECIMAL(10, 2) NOT NULL,
    dataHora DATETIME NOT NULL
);

CREATE TABLE Produto (
    idProduto INT AUTO_INCREMENT PRIMARY KEY,
    valor DECIMAL(10, 2) NOT NULL
);

CREATE TABLE ItemVenda (
    idItemVenda INT AUTO_INCREMENT PRIMARY KEY,
    idVenda INT,
    idProduto INT,
    valorUnitario DECIMAL(10, 2) NOT NULL,
    quantidade INT NOT NULL,
    FOREIGN KEY (idVenda) REFERENCES Venda(idVenda),
    FOREIGN KEY (idProduto) REFERENCES Produto(idProduto),
    UNIQUE (idVenda, idProduto)
);

-- Com base no modelo de Venda & Produto, defina uma consulta SQL que retorne a lista de produtos que participaram de todas as vendas.

SELECT p.idProduto, p.valor
FROM produto AS p
JOIN itemvenda AS iv USING (`idProduto`)
GROUP BY p.idProduto, p.valor
HAVING COUNT(DISTINCT iv.idVenda) = (
    SELECT COUNT(*) FROM venda
);
